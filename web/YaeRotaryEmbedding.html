01 · YaeConfig — คอนฟิกของโมเดล
YaeConfig — ตัวตั้งค่าพารามิเตอร์หลัก

YaeConfig เป็นคลาสเก็บค่าคอนฟิกทั้งหมดของโมเดล YAE ประกอบด้วย:

hidden_size

ขนาดของ hidden state (เช่น 3584 ใน Qwen2.5-7B)

num_attention_heads

จำนวนหัว Attention (เช่น 28 ใน Qwen2.5-7B)

max_position_embeddings

ความยาว Context สูงสุด (เช่น 4096 หรือ 8192)

rope_parameters

พารามิเตอร์ RoPE — rope_type, base ฯลฯ

💡 ใช้ getattr(config, "rope_parameters", {}) or {} — ป้องกันกรณี config ไม่มี field นี้ (safe access)
02 · YaeRotaryEmbedding — หัวใจของ RoPE
Core Logic — การคำนวณ inv_freq
θi = 1 / base(2i/dim)

inv_freq คือ "ความถี่ฐาน" ใช้ในการหมุนเวกเตอร์ตามตำแหน่ง token มิติแรกๆ จะมีความถี่สูง (หมุนเร็ว) ใช้จดจำรายละเอียดคำติดกัน มิติท้ายๆ จะมีความถี่ต่ำ (หมุนช้า) ใช้จดจำบริบทภาพรวม

YaeRotaryEmbedding.py PyTorch + Dynamic RoPE 
class YaeRotaryEmbedding(nn.Module):
    inv_freq: torch.Tensor

    def __init__(self, config: "YaeConfig", device=None):
        super().__init__()
        self.max_seq_len_cached = config.max_position_embeddings
        self.config = config

        # 💡 Safe access — ป้องกัน None
        rope_params = getattr(self.config, "rope_parameters", {}) or {}
        self.rope_type = rope_params.get("rope_type", "default")

        rope_init_fn: Callable = self.compute_default_rope_parameters

        if self.rope_type != "default" and self.rope_type in ROPE_INIT_FUNCTIONS:
            rope_init_fn = ROPE_INIT_FUNCTIONS[self.rope_type]

        inv_freq, self.attention_scaling = rope_init_fn(self.config, device)

        # 🔒 register_buffer = ค่าคงที่ ไม่มี gradient
        self.register_buffer("inv_freq", inv_freq, persistent=False)

    @staticmethod
    def compute_default_rope_parameters(
        config: "YaeConfig",
        device: Optional["torch.device"] = None,
        seq_len: int | None = None,
    ) -> tuple["torch.Tensor", float]:
        rope_params = getattr(config, "rope_parameters", {}) or {}
        base = rope_params.get("rope_parameters", 10000.0)

        dim = getattr(config, "head_dim", None) or config.hidden_size // config.num_attention_heads

        inv_freq = 1.0 / (base ** (torch.arange(0, dim, 2, dtype=torch.int64).to(device=device, dtype=torch.float) / dim))
        return inv_freq, 1.0

    @torch.no_grad()
    def forward(self, x, position_ids):
        inv_freq_expanded = self.inv_freq[None, :, None].float().expand(position_ids.shape[0], -1, 1).to(x.device)
        position_ids_expanded = position_ids[:, None, :].float()

        freqs = (inv_freq_expanded.float() @ position_ids_expanded.float()).transpose(1, 2)
        emb = torch.cat((freqs, freqs), dim=-1)

        cos = emb.cos() * self.attention_scaling
        sin = emb.sin() * self.attention_scaling

        return cos.to(dtype=x.dtype), sin.to(dtype=x.dtype)
03 · rotate_half — หมุนครึ่งมิติสำหรับ RoPE
ทำไมต้อง rotate_half?

RoPE ใช้การหมุนแบบ 2D กับคู่มิติ (x₁, x₂) → (-x₂, x₁) ฟังก์ชันนี้ใช้สำหรับการคำนวณ sine component ในการหมุนเวกเตอร์

rotate_half RoPE Helper 
def rotate_half(x: torch.Tensor) -> torch.Tensor:
    """
    หมุนครึ่งมิติ — (x, y) → (-y, x)
    """
    x1 = x[..., : x.shape[-1] // 2]
    x2 = x[..., x.shape[-1] // 2 :]
    return torch.cat((-x2, x1), dim=-1)
🧮 สูตร RoPE เต็มรูปแบบ: x_rotated = x * cos(θ) + rotate_half(x) * sin(θ)
04 · repeat_kv — Grouped Query Attention (GQA)
GQA — ทำไมต้องมี?

ในโมเดลสมัยใหม่ (Qwen3, LLaMA) จำนวนหัว Query (Q) มักมากกว่า Key/Value (K/V) เพื่อประหยัดหน่วยความจำ repeat_kv ทำหน้าที่ขยาย K/V ให้มีจำนวนหัวเท่ากับ Q โดยใช้การทำซ้ำ (broadcast) แทนการคัดลอกข้อมูลจริง — ประหยัด VRAM สูงสุด

repeat_kv GQA Core 
def repeat_kv(hidden_states: torch.Tensor, n_rep: int) -> torch.Tensor:
    """
    ขยาย K/V heads ให้เท่ากับ Q heads
    ใช้ expand() แทน repeat() — ประหยัดหน่วยความจำ
    """
    batch, num_key_value_heads, slen, head_dim = hidden_states.shape

    if n_rep == 1:
        return hidden_states

    hidden_states = hidden_states[:, :, None, :, :].expand(
        batch, num_key_value_heads, n_rep, slen, head_dim
    )
    return hidden_states.reshape(batch, num_key_value_heads * n_rep, slen, head_dim)
💡 GQA ใน Qwen2.5-7B: Q = 28 heads, K/V = 4 heads, n_rep = 7 ประหยัด VRAM ได้ ~75% เมื่อเทียบกับ Multi-Head Attention แบบเต็ม
05 · eager_attention_forward — Scaled Dot-Product Attention
6 ขั้นตอนของ Attention
eager_attention_forward PyTorch Eager 
def eager_attention_forward(
    module: nn.Module,
    query: torch.Tensor,
    key: torch.Tensor,
    value: torch.Tensor,
    attention_mask: torch.Tensor | None,
    scaling: float,
    dropout: float = 0.0,
):
    # 1. ขยาย K/V heads ให้เท่ากับ Q (GQA)
    n_kv_groups = getattr(module, "num_key_value_groups", 1)
    key_states = repeat_kv(key, n_kv_groups)
    value_states = repeat_kv(value, n_kv_groups)

    # 2. คำนวณ Attention Scores: Q @ K^T และ Scale
    attn_weights = torch.matmul(query, key_states.transpose(2, 3)) * scaling

    # 3. ใช้ Mask (Causal / Padding)
    if attention_mask is not None:
        attn_weights = attn_weights + attention_mask

    # 4. Softmax + Dropout
    attn_weights = nn.functional.softmax(attn_weights, dim=-1, dtype=torch.float32).to(query.dtype)
    attn_weights = nn.functional.dropout(attn_weights, p=dropout, training=module.training)

    # 5. Weighted Sum: Weights @ V
    attn_output = torch.matmul(attn_weights, value_states)

    # 6. Reshape กลับ
    attn_output = attn_output.transpose(1, 2).contiguous()

    return attn_output, attn_weights

Step 1-2

Q @ K^T + Scale

Step 3-4

Mask + Softmax + Dropout

Step 5-6

Weights @ V + Reshape
